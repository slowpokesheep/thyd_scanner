\begin{answer}
  \begin{rail}
    program : ( function + )
  \end{rail}
  \begin{rail}
    function  : NAME '('  ( NAME + ',' ) ? ')' ;
    arrow     : curlL ( decl ';' + ) ? ( expr ';' + ) curlR
  \end{rail}
  \begin{rail}
    decl : 'var' ( NAME + ',' )
  \end{rail}
  \begin{rail}
    expr  : NAME
          | NAME '=' expr
          | NAME '('  ( expr + ',' ) ?   ')'
          | 'return' expr
          | OPNAME expr
          | expr OPNAME expr
          | LITERAL
          | '('  expr  ')'
          | ifexpr
          | 'while' '('  expr  ')' body
  \end{rail}
  \begin{rail}
    ifexpr  : 'if' '('  expr  ')' body ;
    arrow   : (  ( 'elsif' '('  expr  ')'  body  + )  ) ? ;
    arrow   : ( 'else' body ) ? ;
  \end{rail}
  \begin{rail}
    body : curlL ( expr ';' + ) curlR
  \end{rail}
\end{answer}
